def plot_maps(time_value, en=0):

    ni = int(np.where(ds["time"].values == time_value)[0][0])
    ttt = str(time_value.astype("datetime64[s]"))
    fig, (ax1, ax2) = plt.subplots(ncols=2,figsize=(18, 7),subplot_kw={"projection": ccrs.PlateCarree()},constrained_layout=True,)

    # =====================
    # Map 1 (LEFT)
    # =====================
    dta1 = ds['tp'].sel(time=time_value) * 1000
    im1 = ax1.pcolormesh(ds["x"].data, ds["y"].data, dta1, vmin=0,vmax=25, transform=ccrs.PlateCarree(),)

    ax1.coastlines(resolution="50m")
    # ax1.set_title(f"Map 1 – Precip at {stm1}")
    ax1.set_title(f"ERA5-NYC: Precip at {ttt}")

    gl1 = ax1.gridlines(draw_labels=True, linewidth=0.7, color="gray", alpha=0.5, linestyle="--", dms=True,)
    gl1.top_labels = False
    gl1.right_labels = False

    plt.colorbar(im1, ax=ax1, label="Total Precipitation (mm)", shrink=0.6, extend="max",)

    # =====================
    # Map 2 (RIGHT)
    # =====================
    dta2 = ds_nc['tp'].sel(time = time_value, ensemble = en)
    im2 = ax2.pcolormesh(ds_nc["longitude"].data, ds_nc["latitude"].data, dta2, vmin=0, vmax=25, transform=ccrs.PlateCarree(),)

    ax2.coastlines(resolution="50m")
    # ax2.set_title(f"Map 2 – Precip at {stm2} | ens={en}")
    ax2.set_title(f"CorrDiff–NYC: Precip at {ttt} | ens={en}")

    gl2 = ax2.gridlines(draw_labels=True, linewidth=0.7, color="gray", alpha=0.5, linestyle="--", dms=True,)
    gl2.top_labels = False
    gl2.right_labels = False

    plt.colorbar(im2,ax=ax2,label="Total Precipitation (mm)", shrink=0.6, extend="max",)
    plt.show()
